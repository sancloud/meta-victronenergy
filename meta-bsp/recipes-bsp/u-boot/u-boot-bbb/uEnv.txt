# You can hardcode your fdtfile or use some sort of test
# If you do not provide an uEnv.txt, then am335x-boneblack.dtb
# will be the default. (Hard coded in u-boot).
bootpart=1:2
bootdir=/boot
bootfile=zImage
console=ttyO0,115200n8
fdtaddr=0x88000000
fdtfile=bbb-venus.dtb
loadaddr=0x82000000
mmcroot=/dev/mmcblk1p2
mmcrootfstype=ext4 rootwait
optargs=consoleblank=0
mmcargs=setenv bootargs console=${console} ${optargs} root=${mmcroot} ro rootfstype=${mmcrootfstype}
loadfdt=load mmc ${bootpart} ${fdtaddr} ${bootdir}/${fdtfile}
loadimage=load mmc ${bootpart} ${loadaddr} ${bootdir}/${bootfile}
setroot=if test ${version} = 2; then setenv bootpart 1:3; setenv mmcroot /dev/mmcblk1p3; fi
uenvcmd=run setroot; if run loadfdt; then echo Loaded ${fdtfile}; if run loadimage; then run mmcargs; bootz ${loadaddr} - ${fdtaddr}; fi; fi;
